ScanInt64

ScanInt64 can be used to scan int64 and *int64.
Categories:
  • Scanner

{{ ScanInt64 Field }}

type Book struct {
  ID      int64
  Pages   *int64
  Edition sql.Null[int64]
}

type Query struct {
  Title string
}

var queryBook = sqlt.First[Query, Book](sqlt.Parse(`
  SELECT
    id          {{ ScanInt64 "ID" }}
    , pages     {{ ScanInt64 "Pages" }}
    , edition   {{ Scan "Edition" }}
  FROM books
  WHERE title = {{ .Title }}
`))
