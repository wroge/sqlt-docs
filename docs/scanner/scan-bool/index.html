ScanBool

ScanBool can be used to scan bool and *bool.
Categories:
  • Scanner

{{ ScanBool Field }}

type Favorite bool

type Book struct {
  Favorite Favorite
  Read     bool
}

type Query struct {
  Title string
}

var queryBook = sqlt.First[Query, Book](sqlt.Parse(`
  SELECT
    true    {{ ScanBool "Favorite" }}
    , read  {{ ScanBool "Read" }}
  FROM books
  WHERE title = {{ .Title }}
`))
