ScanUint

ScanUint can be used to scan uints and pointer to uints.
Executors:
  • First
Scanners:
  • Scan
  • ScanUint

{{ ScanUint Field }}

Example
type Book struct {
  ID      uint
  Pages   *uint64
  Edition sql.Null[uint64]
}

type Query struct {
  Title string
}

var queryBook = sqlt.First[Query, Book](sqlt.Parse(`
  SELECT
    id          {{ ScanUint "ID" }}
    , pages     {{ ScanUint "Pages" }}
    , edition   {{ Scan "Edition" }}
  FROM books
  WHERE title = {{ .Title }}
`))
